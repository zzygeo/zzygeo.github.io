写一些觉得能用上的C++代码：

将文本的数字读成数组： 

#include<iostream>
#include<fstream>
using namespace std;
int main()
{
    const int txtRow = 201;
    const int txtColumn = 8;
    float txtData[txtRow][txtColumn];
    ifstream infile;
    infile.open("C:\\APP\\C\\Project2\\测试数据.txt");
    for (int i = 0; i < txtRow; i++)
    {
        for (int j = 0; j < txtColumn; j++)
        {
            infile >> txtData[i][j];
        }
    }
    infile.close();

return 0;
}

将文本的数字读成矩阵，需要第三方库Eigen。 

#include<iostream>
#include<Eigen/Dense>
#include<fstream>
#include<cmath>
using namespace std;
using namespace Eigen;

MatrixXd ReadData(istream& data, int a, int b)
{
    MatrixXd m_matrix(a, b);
    for (int i = 0; i < a; i++)
    {
        for (int j = 0; j < b; j++)
        {
            data >> m_matrix(i,j);
        }
    }
    return m_matrix;
}

}

冒泡排序：

#include<iostream>
#include<vector>
using namespace std;

void display(vector<int> vec)
{
    for (int i = 0; i < vec.size(); ++i)
        cout << vec[i] << ' ';
    cout << endl;
}

void swap(int& val1, int& val2)
{
    int temp = val1;
    val1 = val2;
    val2 = temp;
}

void bubble_sort(vector<int>& vec)
{
    for (int ix = 0; ix < vec.size(); ++ix)
        for (int jx = ix + 1; jx < vec.size(); ++jx)
            if (vec[ix] > vec[jx])
                swap(vec[ix], vec[jx]);
}

int main()
{
    int al[8] = { 2,56,4,5,13,7,1,24 };
    vector<int> vec(al, al + 8);
    cout << "before sort:  ";
    display(vec);
    bubble_sort(vec);
    cout << "after sort: ";
    display(vec);

    return 0;
}

关键在于swap 和 bubble_sort时，不仅要传值，也要传址，不然只是其值被复制了一份，成为了参数的局部性定义，当运行结束也就从“程序堆栈”里释放掉了，并没有关联起来。
